محاسبه فاکتوریل

این برنامه یک عدد از کاربر دریافت می‌کند و فاکتوریل آن را محاسبه می‌کند.
def factorial(n): if n == 0 or n == 1: return 1 else: return n * factorial(n-1) # دریافت ورودی از کاربر number = int(input("لطفاً یک عدد وارد کنید: ")) result = factorial(number) print(f"فاکتوریل {number} برابر است با: {result}")

تشخیص عدد اول

این برنامه بررسی می‌کند که آیا عدد وارد شده اول است یا خیر.
def is_prime(n): if n <= 1: return False if n <= 3: return True if n % 2 == 0 or n % 3 == 0: return False i = 5 while i * i <= n: if n % i == 0 or n % (i + 2) == 0: return False i += 6 return True # دریافت ورودی از کاربر number = int(input("لطفاً یک عدد وارد کنید: ")) if is_prime(number): print(f"{number} یک عدد اول است.") else: print(f"{number} یک عدد اول نیست.")

تحلیل احساسات متن فارسی

این برنامه با استفاده از کتابخانه PerSent، احساسات متن فارسی را تحلیل می‌کند.
# شبیه‌سازی کتابخانه PerSent برای تحلیل احساسات def analyze_sentiment(text): positive_words = ['خوب', 'عالی', 'عالیه', 'زیبا', 'شادی', 'خوشحال', 'مثبت', 'عالیست'] negative_words = ['بد', 'زشت', 'ناراحت', 'غم', 'منفی', 'بدی', 'مشکل'] words = text.split() positive_count = sum(1 for word in words if word in positive_words) negative_count = sum(1 for word in words if word in negative_words) if positive_count > negative_count: return "مثبت", positive_count / len(words) * 100 elif negative_count > positive_count: return "منفی", negative_count / len(words) * 100 else: return "خنثی", 50 # دریافت ورودی از کاربر text = input("لطفاً متن فارسی خود را وارد کنید: ") sentiment, confidence = analyze_sentiment(text) print(f"احساسات متن: {sentiment}") print(f"درصد اطمینان: {confidence:.2f}%")

مرتب‌سازی لیست اعداد

این برنامه یک لیست از اعداد را از کاربر دریافت کرده و به صورت صعودی مرتب می‌کند.
def bubble_sort(numbers): n = len(numbers) for i in range(n): for j in range(0, n-i-1): if numbers[j] > numbers[j+1]: numbers[j], numbers[j+1] = numbers[j+1], numbers[j] return numbers # دریافت ورودی از کاربر input_str = input("لطفاً اعداد را با کاما جدا کنید: ") numbers = [int(x.strip()) for x in input_str.split(',')] sorted_numbers = bubble_sort(numbers) print(f"لیست مرتب شده: {sorted_numbers}")
